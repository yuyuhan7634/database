@Transactional(rollbackFor = Exception.class)
    public boolean approveCheckout(String applyNo) {
        Application application = applicationMapper.selectById(applyNo);
        if (application == null || application.getApplyStatus() != 0) {
            throw new RuntimeException("退房申请单不存在或已被处理！");
        }
        String ownerName = application.getApplicantName();
        Resident resident = residentMapper.selectById(ownerName);
        if (resident == null) {
            throw new RuntimeException("找不到该住户的在住信息，无法退房！");
        }

        String houseNo = resident.getHouseNo();
        residentMapper.deleteById(ownerName);
        House house = houseMapper.selectById(houseNo);
        if (house != null) {
            house.setStatus(0);
            houseMapper.updateById(house);
        }
        application.setApplyStatus(1);
        applicationMapper.updateById(application);

        System.out.println("====== 已为 [" + ownerName + "] 办理退房，房产 [" + houseNo + "] 已回收 ======");
        return true;
    }
    // ================== 3. 调房审批 ==================
    @Transactional(rollbackFor = Exception.class)
    public boolean approveTransfer(String applyNo, String newHouseNo) {
        Application application = applicationMapper.selectById(applyNo);
        if (application == null || application.getApplyStatus() != 0) {
            throw new RuntimeException("调房申请单不存在或已被处理！");
        }
        String ownerName = application.getApplicantName();
        Resident resident = residentMapper.selectById(ownerName);
        if (resident == null) {
            throw new RuntimeException("找不到该住户的原住房信息！");
        }
        String oldHouseNo = resident.getHouseNo();
        House newHouse = houseMapper.selectById(newHouseNo);
        if (newHouse == null || newHouse.getStatus() != 0) {
            throw new RuntimeException("目标新房产不存在或已被占用！");
        }
        House oldHouse = houseMapper.selectById(oldHouseNo);
        if (oldHouse != null) {
            oldHouse.setStatus(0);
            houseMapper.updateById(oldHouse);
        }
        newHouse.setStatus(1);
        houseMapper.updateById(newHouse);
        resident.setHouseNo(newHouseNo);
        resident.setHouseArea(newHouse.getArea());
        residentMapper.updateById(resident);
        application.setApplyStatus(1);
        applicationMapper.updateById(application);
        System.out.println("====== 已为 [" + ownerName + "] 办理调房：[" + oldHouseNo + "] -> [" + newHouseNo + "] ======");
        return true;
    }
}

这段调房核心算法可以拆解为四个核心阶段

1.拦截与身份验真

在正式找房之前，算法做了防伪校验：确保这是一张真实存在的“调房单”，且还在排队中。算法查了这个人，还查了他现在住的房子。

2.采用贪心算法思想，分为两步走：

只找 `status = 0`（空闲）且 `area >= 目标面积` 的房子。小于目标面积的直接淘汰。

定义“好房优先级”：

第一优先级（面积）：`orderByDesc("area")`，在满足条件的范围里，面积越大的越排在前面（优先给大房子）。

第二优先级（租金）：`orderByAsc("rent_per_sqm")`，如果面积一样大，单价越便宜的越排在前面（优先给性价比高的）。

第三优先级（房号）：`orderByAsc("house_no")`，如果面积和单价都一样，按房号从小到大排。

通过这条链式查询，数据库返回的 `availableHouses` 列表，

算法拿着排序好的清单，从最好的一套开始往下遍历。每次遍历，都会去 `house_standard` 表里查这套房的“门槛分”。只要发现申请人的分数（`score`）大于等于这套房的门槛分，立刻锁定这套房（`selectedHouse = h`）并终止循环。

3.找到新房后，将旧房状态立刻改回 `0`，将新选中的好房状态改为 `1`。在 `Resident` 表中抹去旧房信息，写入新房号和新面积，并同步他最新的职称和分数。

4.遵循题目要求，将旧房遗留的关联房租账单全部清理。底层立刻调用 `selectedHouse.getArea().multiply(rent_per_sqm)`生成该住户在新房的首月租金账单。将申请单标记为已处理，并打印出包含“房号变更轨迹”和“新房租金”的调房分配单。
