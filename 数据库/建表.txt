CREATE TABLE house_standard (
    standard_no VARCHAR(20) PRIMARY KEY,
    area DECIMAL(10, 2) NOT NULL UNIQUE,
    min_score INT NOT NULL
);

CREATE TABLE house (
    house_no VARCHAR(50) PRIMARY KEY,
    area DECIMAL(10, 2) NOT NULL,
    status INT NOT NULL,
    rent_per_sqm DECIMAL(10, 2) NOT NULL
);

CREATE TABLE resident (
    owner_name VARCHAR(50) PRIMARY KEY,
    department VARCHAR(100) NOT NULL,
    title VARCHAR(50) NOT NULL,
    family_size INT NOT NULL,
    score INT NOT NULL,
    house_no VARCHAR(50) NOT NULL,
    house_area DECIMAL(10, 2) NOT NULL,
    FOREIGN KEY (house_no) REFERENCES house(house_no)
);

CREATE TABLE rent_bill (
    house_no VARCHAR(50) NOT NULL,
    owner_name VARCHAR(50) NOT NULL,
    bill_month VARCHAR(10) NOT NULL,
    rent_amount DECIMAL(10, 2) NOT NULL,
    create_time DATE,
    PRIMARY KEY (house_no, owner_name, bill_month),
    FOREIGN KEY (house_no) REFERENCES house(house_no),
    FOREIGN KEY (owner_name) REFERENCES resident(owner_name)
);
CREATE TABLE application (
    apply_no VARCHAR(50) PRIMARY KEY,
    applicant_name VARCHAR(50) NOT NULL,
    department VARCHAR(100) NOT NULL,
    apply_type INT NOT NULL,
    apply_status INT NOT NULL,
    req_area DECIMAL(10, 2),
    old_house_no VARCHAR(50),
    create_time DATE
);
